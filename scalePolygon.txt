LEXICON
========
inpoly: the modified polygon with the new point
factor: float wich is the coefficient of expension relative to a reference point (we take the head for the reference point)
i: integer
referencePoint : PointElement wich keep the initial head

ALGORITHM
=========
Data:   Polygon: inpoly
        Float: factor
Return: inpoly


function scalePolygon(inpoly : Polygon, factor : Float) : Polygon
Begin
	
	referencePoint <- head(inpoly)

	for i = 1 to size(inpoly) - 1 do
		head(inpoly) <- next(head(inpoly))

		x(value(head(inpoly)) = x(value(head(inpoly)) + factor * (x(value(head(inpoly)) - x(value(referencePoint)))
		y(value(head(inpoly)) = y(value(head(inpoly)) + factor * (y(value(head(inpoly)) - y(value(referencePoint)))

	done

	head(inpoly) = referencePoint

	scalePolygon <- inpoly
End
