LEXICON
========
newpoly: the modified polygon with the new point
newelem: the new element which will be added

ALGORITHM
=========
Data: 	Polygon: inpoly
	Point: inpoint
Return: newpoly


function addPoint(inpoly : Polygon, inpoint : Point) : Polygon
Begin

	newpoly <- inpoly
	value(newelem) <- inpoint
	
	if inpoly==NULL then do

		newpoly <- newelem
		next(newpoly) <- newpoly
		previous(newpoly) <- newpoly

	else

		next(newelem) <- inpoly
		previous(newelem) <- previous(inpoly)
		next(previous(newpoly) <- newelem
		previous(newpoly) <- newelem

	done

	addPoint <- newpoly

End
